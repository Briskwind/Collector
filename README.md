```
python3.5

```

#### web demo 小栗子

#### gunicorn 启动web
```bash

# 启动worker
celery -A celery_pro worker -l info -c 1
# 启动beat,可以在启动worker中增加-B 参数替代
celery -A celery_pro beat  -l info 

Collector/deploy/gunicorn
pm2 start web.sh
```
